version: '3'
services:
  
  rabbitmq:
    image: "rabbitmq:3-management"
    container_name: "rabbitmq"
    ports:
      - "15672:15672"  # Panel de administración de RabbitMQ
      - "5672:5672"    # Puerto para comunicación con los productores/consumidores
    environment:
      RABBITMQ_DEFAULT_USER: user
      RABBITMQ_DEFAULT_PASS: password

  producer1:
    build: ./producer  # Usa el código del productor
    depends_on:
      - rabbitmq
    environment:
      RABBITMQ_HOST: rabbitmq
      RABBITMQ_PORT: 5672

  producer2:
    build: ./producer  # Usa el mismo código para el segundo productor
    depends_on:
      - rabbitmq
    environment:
      RABBITMQ_HOST: rabbitmq
      RABBITMQ_PORT: 5672

  consumer1:
    build: ./consumer  # Usa el código del consumidor
    depends_on:
      - rabbitmq
    environment:
      RABBITMQ_HOST: rabbitmq
      RABBITMQ_PORT: 5672

  consumer2:
    build: ./consumer  # Usa el mismo código para el segundo consumidor
    depends_on:
      - rabbitmq
    environment:
      RABBITMQ_HOST: rabbitmq
      RABBITMQ_PORT: 5672
